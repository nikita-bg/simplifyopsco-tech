**1.2 API Key Generation System**
```typescript
// Generate unique API key per business on signup
import { nanoid } from 'nanoid';
import crypto from 'crypto';

export async function generateApiKey(businessId: string) {
  // Public key (client-side, read-only)
  const publicKey = `so_pub_${nanoid(32)}`;

  // Secret key (server-side only, full access)
  const secretKey = `so_sec_${nanoid(32)}`;

  // Hash secret key before storing
  const secretHash = crypto
    .createHash('sha256')
    .update(secretKey)
    .digest('hex');

  await supabase.from('businesses').update({
    public_key: publicKey,
    secret_key_hash: secretHash,
  }).eq('id', businessId);

  return { publicKey, secretKey };
}
```

#### 4.3.2 CORS & Origin Validation

**Problem:** WebSocket connections from ANY domain (security hole!)

**Solution: Domain Whitelisting**
```typescript
// Validate origin in WebSocket handshake
import { WebSocketServer } from 'ws';

wss.on('connection', async (ws, req) => {
  const origin = req.headers.origin;
  const apiKey = req.url.split('api_key=')[1];

  // Fetch allowed domains for this business
  const { data: business } = await supabase
    .from('businesses')
    .select('allowed_domains, public_key')
    .eq('public_key', apiKey)
    .single();

  if (!business) {
    ws.close(4401, 'Invalid API key');
    return;
  }

  // Check if origin is whitelisted
  const allowedDomains = business.allowed_domains || [];
  const isAllowed = allowedDomains.some(domain =>
    origin?.includes(domain)
  );

  if (!isAllowed && process.env.NODE_ENV === 'production') {
    ws.close(4403, 'Origin not whitelisted');
    return;
  }

  // Connection approved
  // ...
});
```

#### 4.3.3 Rate Limiting per Widget

**Problem:** Malicious customer embeds 1000 widgets to DDoS attack

**Solution: Multi-Layer Rate Limiting**
```typescript
import rateLimit from 'express-rate-limit';
import RedisStore from 'rate-limit-redis';

// Layer 1: Global rate limit (all requests)
const globalLimiter = rateLimit({
  store: new RedisStore({ client: redisClient }),
  windowMs: 15 * 60 * 1000,  // 15 minutes
  max: 1000,  // 1000 requests per 15 min globally
  message: 'Too many requests, please try again later'
});

// Layer 2: Per-business rate limit
const businessLimiter = rateLimit({
  store: new RedisStore({ client: redisClient }),
  windowMs: 60 * 60 * 1000,  // 1 hour
  max: (req) => {
    const business = req.business;  // From API key middleware
    return business.plan_tier === 'free' ? 100 :
           business.plan_tier === 'starter' ? 500 :
           business.plan_tier === 'pro' ? 2000 : 10000;
  },
  keyGenerator: (req) => req.business.id,
  message: 'Business rate limit exceeded'
});

// Layer 3: Concurrent connections limit
const activeConnections = new Map<string, number>();

function checkConcurrentLimit(businessId: string, limit: number): boolean {
  const current = activeConnections.get(businessId) || 0;
  if (current >= limit) {
    return false;  // Reject connection
  }
  activeConnections.set(businessId, current + 1);
  return true;
}
```

#### 4.3.4 Public/Private Key Separation

**Problem:** API key is client-side visible in page source!

**Solution: Two-Tier Key System**
```html
<!-- Customer embeds PUBLIC key only (read-only) -->
<script>
  window.SimplifyOpsConfig = {
    apiKey: 'so_pub_abc123...',  // ✅ Safe to expose
    position: 'bottom-right'
  };
</script>

<!-- Server-to-server uses SECRET key (full access) -->
```

**Key Permissions:**
| Key Type | Can Do | Cannot Do |
|----------|--------|-----------|
| **Public** (so_pub_...) | Connect widget, send voice, receive responses | Access dashboard, modify settings, view analytics |
| **Secret** (so_sec_...) | Full API access, modify business settings | Nothing - has full access |

**Implementation:**
```typescript
async function validateApiKey(key: string): Promise<{business: Business, keyType: 'public' | 'secret'}> {
  if (key.startsWith('so_pub_')) {
    // Public key: Widget-only access
    const business = await supabase
      .from('businesses')
      .select('*')
      .eq('public_key', key)
      .single();
    return { business, keyType: 'public' };
  } else if (key.startsWith('so_sec_')) {
    // Secret key: Full access (hash and compare)
    const hash = crypto.createHash('sha256').update(key).digest('hex');
    const business = await supabase
      .from('businesses')
      .select('*')
      .eq('secret_key_hash', hash)
      .single();
    return { business, keyType: 'secret' };
  }
  throw new Error('Invalid API key format');
}
```

    // Calculate overage
    const overage = current - limit;
    const overageCost = overage * OVERAGE_RATES[business.plan_tier];

    // Create invoice item (Stripe)
    await stripe.invoiceItems.create({
      customer: business.stripe_customer_id,
      amount: Math.round(overageCost * 100),  // Cents
      currency: 'usd',
      description: `Overage: ${overage} conversations`
    });
  }

  // 3. Hard limit (prevent abuse)
  if (current > limit * 2) {
    throw new Error('Conversation limit exceeded. Please upgrade.');
  }
}
