```python
row = await db.fetchrow(
    "SELECT subscription_tier, minutes_used FROM stores WHERE id = $1::uuid", store_id
)
if row:
    from backend.stripe_service import TIER_LIMITS
    limit = TIER_LIMITS.get(row["subscription_tier"] or "trial", 30)
    if (row["minutes_used"] or 0) >= limit * 1.1:
        await db.execute(
            "UPDATE stores SET agent_status = 'limit_reached' WHERE id = $1::uuid AND agent_status = 'active'",
            store_id
        )
        SecurityLogger.log(f"Usage limit reached: store={store_id} minutes={row['minutes_used']}/{limit}", "WARNING")
```
